
public class TriangleStu {
	public static void Triangle(int n){
	    for(int i=1;i<=n;i++){
	        for(int j=1;j<=i;j++){
	            System.out.print(j+" ");
	        }
	        System.out.print("\n");
	    }
		
	}
	public static void main(String[] args) {
	    Triangle(5);
		
	}
}
